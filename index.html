async function handleFileSync(req, res) {
  const { userId, fileId } = req.params;

  // Authorize and fetch in one query — no extra round trip
  const file = await db.files.findFirst({
    where: { id: fileId, ownerId: userId },
    select: { id: true, checksum: true, size: true }
  });

  if (!file) return res.status(404).json({ error: "File not found" });

  const job = await queue.add("file-sync", {
    fileId:   file.id,
    checksum: file.checksum,
    priority: file.size < 1_000_000 ? "high" : "normal"
  });

  return res.status(202).json({ jobId: job.id, status: "queued" });
}
function useRealTimeMetrics(projectId) {
  const [metrics, setMetrics] = useState(null);
  const [error,   setError]   = useState(null);

  useEffect(() => {
    const ws = new WebSocket(
      `wss://api.devboard.io/metrics/${projectId}`
    );

    ws.onmessage = (e) => {
      const update = JSON.parse(e.data);
      setMetrics(prev => ({ ...prev, ...update }));
    };

    ws.onerror = () => setError(new Error("Connection lost"));

    return () => ws.close(); // always clean up
  }, [projectId]);

  return { metrics, error, isLoading: !metrics && !error };
}
describe("handleFileSync", () => {
  const mockFile = { id: "f-1", checksum: "abc123", size: 500_000 };

  it("queues a high-priority job for files under 1MB", async () => {
    db.files.findFirst.mockResolvedValue(mockFile);
    queue.add.mockResolvedValue({ id: "job-1" });

    const res = await request(app)
      .get("/files/user-1/f-1/sync")
      .expect(202);

    expect(res.body.jobId).toBe("job-1");
    expect(queue.add).toHaveBeenCalledWith(
      "file-sync",
      expect.objectContaining({ priority: "high" })
    );
  });

  it("returns 404 when file does not belong to user", async () => {
    db.files.findFirst.mockResolvedValue(null);
    await request(app)
      .get("/files/user-1/f-999/sync")
      .expect(404);
  });
});
